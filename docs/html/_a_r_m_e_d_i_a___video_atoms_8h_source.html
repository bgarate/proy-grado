78 /* Atoms :
79 LEGEND :
80 -> specific = specific function to generate this atom
81 -> empty = use atomFromData (0, "name", NULL);
82 -> from data = use atomFromData (dataSize, "name", dataPointer);
83 
84 ftyp -> specific
85 pvat -> specific
86 mdat -> specific (include freeAtom if needed)
87 moov -> empty
88  |- mvhd -> specific
89  |- trak -> empty
90  | |- tkhd -> specific
91  | \- mdia -> empty
92  | |- mdhd -> specific
93  | |- hdlr -> specific (for mdia)
94  | \- minf -> empty
95  | |- vmhd -> specific
96  | |- hdlr -> specific (for minf)
97  | |- dinf -> empty
98  | | \- dref -> specific
99  | \- stbl -> empty
100  | |- stsd -> specific
101  | |- stts -> from data
102  | |- stss -> from data (i frame positions as uint32_t network endian)
103  | |- stsc -> specific
104  | |- stsz -> from data (frames sizes as uint32_t network endian)
105  | \- stco -> from data (frames offset as uint32_t network endian)
106 */
